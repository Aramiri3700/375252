    video { transform: scaleX(-1); width: 100%; height: 100%; object-fit: contain; }
    canvas { position: absolute; transform: scaleX(-1); }
    #ui-layer { position: fixed; bottom: 50px; width: 100%; z-index: 10; }

  <div id="container">
    <video id="webcam" autoplay playsinline></video>
    <canvas id="output_canvas"></canvas>
  </div>
  <div id="ui-layer">
    <button id="startBtn">システムを起動する</button>
  </div>
  <script>
    const video = document.getElementById('webcam');
    const canvas = document.getElementById('output_canvas');
    const ctx = canvas.getContext('2d');
    const startBtn = document.getElementById('startBtn');
    const SOUND_URL = 'https://drive.google.com/uc?export=download&id=1D4DhwxdKXVJb-u18SEVX9XhfZ2fdjEVZ';
    const detectSound = new Audio(SOUND_URL);
    let faceDetection;
    let lastDetected = false;
    async function init() {
      faceDetection = new FaceDetection({
        locateFile: (file) => `https://cdn.jsdelivr.net/npm/@mediapipe/face_detection/${file}`
      });
      faceDetection.setOptions({ model: 'short', minDetectionConfidence: 0.6 });
      faceDetection.onResults(onResults);
      const camera = new Camera(video, {
        onFrame: async () => { await faceDetection.send({image: video}); },
        width: 1280, height: 720, facingMode: "user"
      });
      camera.start();
    }
    function onResults(results) {
      canvas.width = video.clientWidth;
      canvas.height = video.clientHeight;
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      if (results.detections.length > 0) {
        if (!lastDetected) {
          detectSound.currentTime = 0;
          detectSound.play().catch(e => console.error(e));
          lastDetected = true;
        }
      } else { lastDetected = false; }
    }
    startBtn.addEventListener('click', () => {
      detectSound.play().then(() => { detectSound.pause(); });
      init();
      startBtn.style.display = 'none';
    });
  </script>
